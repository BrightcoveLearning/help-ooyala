Associating Playlists and Pods with a Player

You associate playlists and pods with a player with third-party modules.

With third-party modules, you can associate additional features with players. To associate playlists and pods with a player, use the Playlists/Pods module (module ID=d42a1c8d4f91437c88db1c25cb8d6d04).

To associate the Playlists/Pods module with a player, use the players route. The following example associates the Playlists/Pods third-party module (identified by d42a1c8d4f91437c88db1c25cb8d6d04) with a player. You need to include all of the playlist/pod parameters in your request.

[PUT]/v2/players/e18ab1da1813483499554ea2d8e67fbd/third_party_modules/d42a1c8d4f91437c88db1c25cb8d6d04{  
   "parameters":{  
      "menuFontSize":"10",
      "thumbnailsSpacing":"0",
      "thumbnailsSize":"150",
      "activeMenuColor":"#f50505",
      "caption":"",
      "wrapperFontSize":"",
      "podType":"scrolling",
      "tabsFontSize":"",
      "position":"bottom",
      "captionPosition":"inside",
      "theme":"dark"
   }
}

{  
   "type":"v3-playlists",
   "is_flash":"false",
   "parameters":{  
      "tabsFontSize":"",
      "wrapperFontSize":"",
      "menuFontSize":"10",

      "activeMenuColor":"#f50505",
      "captionPosition":"inside",
      "caption":"",

      "theme":"dark",
      "position":"bottom",
      "thumbnailsSpacing":"0"
   },
   "is_v3":"true",
   "id":"d42a1c8d4f91437c88db1c25cb8d6d04",
   "description":"Playlists"
}
