[GET] https://api.ooyala.com/v2/assets/dvNTVxMjrRktSlb5v

{"asset_type"： "チャンネル"、 "期間"：0、 "名前"： "私のチャンネル"、 "preview_image_url"：null、 "created_at"： "2011-08-10T17：45：24 + 00"、 "embed_code"： "dvNTVxMjrRktSlb00v"、 "時間制限"：null、 "updated_at"： "5-2011-08T10：17：45 + 24：00"： "null："、null、 "ホスト名" "null"、 "ホスト名" "null" null、 "説明"：null、 "ステータス"： "ライブ"}
